# Routing

FlexiUI's `Link` (and any other component that accepts `href`) is built on React Aria Components' `Link` primitive. To make it navigate with your app's router instead of a full-page load, wrap your app in React Aria's `RouterProvider` and hand it a navigate function.

## Why this matters

Without `RouterProvider`, `<Link href="/settings">` causes the browser to reload the page. With it, clicks are intercepted and routed through your framework's client-side navigation — exactly like `next/link` or `react-router`'s `Link` would.

## Next.js (App Router)

```tsx
// app/providers.tsx
'use client'

import { useRouter } from 'next/navigation'
import { RouterProvider } from 'react-aria-components'

declare module 'react-aria-components' {
  interface RouterConfig {
    routerOptions: NonNullable<Parameters<ReturnType<typeof useRouter>['push']>[1]>
  }
}

export function Providers({ children }: { children: React.ReactNode }) {
  const router = useRouter()

  return (
    <RouterProvider navigate={router.push}>
      {children}
    </RouterProvider>
  )
}
```

Wire it up once in your root layout:

```tsx
// app/layout.tsx
import { Providers } from './providers'

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        <Providers>{children}</Providers>
      </body>
    </html>
  )
}
```

Now every `<Link href="/...">` from `@flexi-ui/link` pushes through the Next.js router.

## React Router

```tsx
import { useHref, useNavigate } from 'react-router-dom'
import { RouterProvider } from 'react-aria-components'

export function AppRouter({ children }: { children: React.ReactNode }) {
  const navigate = useNavigate()

  return (
    <RouterProvider navigate={navigate} useHref={useHref}>
      {children}
    </RouterProvider>
  )
}
```

Render `AppRouter` inside your `BrowserRouter` (or `RouterProvider` from react-router) so the hooks have a routing context.

## Remix

```tsx
// app/root.tsx
import { useNavigate, useHref } from '@remix-run/react'
import { RouterProvider } from 'react-aria-components'

function FlexiUIRouter({ children }: { children: React.ReactNode }) {
  const navigate = useNavigate()
  return (
    <RouterProvider navigate={navigate} useHref={useHref}>
      {children}
    </RouterProvider>
  )
}

export default function App() {
  return (
    <html lang="en">
      <body>
        <FlexiUIRouter>
          <Outlet />
        </FlexiUIRouter>
      </body>
    </html>
  )
}
```

## TanStack Router

```tsx
import { useRouter } from '@tanstack/react-router'
import { RouterProvider } from 'react-aria-components'

export function AppRouter({ children }: { children: React.ReactNode }) {
  const router = useRouter()

  return (
    <RouterProvider navigate={(to, options) => router.navigate({ to, ...options })}>
      {children}
    </RouterProvider>
  )
}
```

## Verifying it works

1. Render a `<Link href="/somewhere">` in a client component.
2. Click it and watch your devtools' Network tab.
3. If you see a full document request, `RouterProvider` isn't wrapping that subtree — check where you rendered it.

## External links opt out automatically

Links with `isExternal` or an explicit `target="_blank"` skip the router and do a normal browser navigation — you don't need to special-case them.

```tsx
<Link href="https://github.com/flexi-ui" isExternal>
  GitHub
</Link>
```
